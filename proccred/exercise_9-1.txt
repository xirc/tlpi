a) setuid(2000);
# 特権が破棄される
<<<  r=1000, e=0, s=0, f=0
>>>  r=2000, e=2000, s=2000, f=2000

b) setreuid(-1, 2000);
<<< r=1000, e=0, s=0, f=0
>>> r=1000, e=2000, s=0, f=2000

c) seteuid(2000)
<<< r=1000, e=0, s=0, f=0
>>> r=1000, e=2000, s=0, f=2000

d) setfsuid(2000)
<<< r=1000, e=0, s=0, f=0
>>> r=1000, e=0, s=0, f=2000

e) setresuid(-1, 2000, 3000)
<<< r=1000, e=0, s=0, f=0
>>> r=1000, e=2000, s=3000, f=2000
